\newglossaryentry{accesspoint}{
    name={Access Point},
    description={Ein Gerät, das ein WLAN-Signal bereitstellt und den drahtlosen Zugriff auf ein Netzwerk ermöglicht}
}

\newglossaryentry{bssid}{
    name={BSSID},
    description={Die MAC-Adresse eines Access Points in einem WLAN-Netzwerk}
}

\newglossaryentry{ssid}{
    name={SSID},
    description={Der Name eines WLAN-Netzwerks, der zur Identifizierung dient}
}
